Существует интересная возможность использования объекта внутри своего родного класса. Для этого необходимо, чтобы метод, в котором объявляется объект, был static. Рассмотрим пример ниже: 

    
    // Создаем объект внутри своего родного класса
    class Post
    {
    public $title;
    public $body;

    public function __construct($title,  $body)
    {
        $this->title = $title;
        $this->body = $body;
    }
    public static function all()
    {
        $vartitle = "first post";
        $varBody = "first post about OOP";
        return new Post(
            $vartitle,
            $varBody
        );
    }
    }
    echo Post::all()->title . "
"; // first post
    echo Post::all()->body; // first post about OOP
